private alias BGAnimation_value0 : BGAnimation_currentPreview

event ObjectMain
	BGAnimation.BGTimer++
	BGAnimation.BGTimer &= 511
	Sin(temp0, BGAnimation.BGTimer)
	if temp0 > 0
		if temp0 < 32
			screen.yoffset = 256
		else
			screen.yoffset = 0
		end if
	else
		if temp0 > -32
			screen.yoffset = 512
		else
			screen.yoffset = 768
		end if
	end if
end event




event ObjectStartup

	LoadSpriteSheet("LevelSelect/Icons.gif")

	// Set all BG Animation objects to active
	// (There should only ever be one in a scene, though)
	foreach (TypeName[BG Animation], arrayPos0, ALL_ENTITIES)
		object[arrayPos0].priority = PRIORITY_ACTIVE
	next

end event

event RSDKDraw
	DrawSprite(0)
end event

end event
