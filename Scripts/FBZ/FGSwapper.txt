event ObjectDraw
if stage.actnum == 1
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		if Object.PropertyValue==0
			if object[currentPlayer].xpos >= Object.xPos
				Stage.ActiveLayer[3] = 3
			else
				Stage.ActiveLayer[3] = 0
			end if
		else
			if object[currentPlayer].xpos >= Object.xPos
				Stage.ActiveLayer[3] = 0
			else
				Stage.ActiveLayer[3] = 3
			end if
		end if
	next
else 
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		if Object.PropertyValue==0
			if object[currentPlayer].xpos >= Object.xPos
				Stage.ActiveLayer[3] = 4
			else
				Stage.ActiveLayer[3] = 0
			end if
		else
			if object[currentPlayer].xpos >= Object.xPos
				Stage.ActiveLayer[3] = 0
			else
				Stage.ActiveLayer[3] = 4
			end if
		end if
		next
end if
end event
