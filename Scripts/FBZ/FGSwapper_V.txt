
//-------Aliases-------//
#alias 35: TYPE_BGSWAPPER


event ObjectDraw
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		if Object.PropertyValue==0
			if object[currentPlayer].ypos >= Object.YPos
				Stage.ActiveLayer[3] = 3
			else
				Stage.ActiveLayer[3] = 0
			end if
		else
			if object[currentPlayer].ypos >= Object.yPos
				Stage.ActiveLayer[3] = 0
			else
				Stage.ActiveLayer[3] = 3
			end if
		end if
	next
end event


event ObjectStartup
	Temp0 = 16384
	Temp1 = 0
	ArrayPos0 = 32
	foreach (TypeName[FGSwapper], ArrayPos0, ALL_ENTITIES)
		if Object[ArrayPos0].XPos < Temp0
				Temp0 = Object[ArrayPos0].XPos
			end if
			if Object[ArrayPos0].XPos > Temp1
				Temp1 = Object[ArrayPos0].XPos
			end if
		end if
		ArrayPos0++
	next
end event

event RSDK
	LoadSpriteSheet("Global/Display.gif")
	SetEditorIcon(Icon0,SingleIcon,-16,-16,32,32,1,143)
end event
